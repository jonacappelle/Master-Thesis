178 void CheckIMUidle( void )
179 {
180 
181  /* Read battery in percent */
182  ADC_get_batt(data.batt);
183 
184  float mean_gyro = (uint8_t) ( data.ICM_20948_gyro[0] + data.ICM_20948_gyro[1] + data.ICM_20948_gyro[2] ) / 3;
185 
186  /* Add 1 sec to the count of idle seconds */
187  if( mean_gyro < 0.1 )
188  {
189  idle_count++;
190  }else{ /* If there is movement, reset idle seconds */
191  idle_count = 0;
192  }
193 
194 
195  if( (idle_count > 60) || (ble_sec_not_connected > 5*5)) // 5sec * 75Hz
196  {
197  idle_count = 0;
198  ble_sec_not_connected = 0;
199  teller_accuracy = 0;
200  beta = 1.0f;
201  /* Dont't check idle state in sleep */
202  RTCDRV_StopTimer( IMU_Idle_Timer );
203  /* Stop generating interrupts */
204  ICM_20948_interruptEnable(false, false);
205  appState = SLEEP;
206  _sleep = true;
207  }
208  // Reset timer is BLE is connected
209  if(bleConnected)
210  {
211  ble_sec_not_connected = 0;
212  }
213 
214 /* For visual representation where in the code */
215 #if DIY == 0
216  BSP_LedToggle(1);
217 #endif
218 
219  if(teller_accuracy < 10)
220  {
221  teller_accuracy++;
222  }
223 
224  if(teller_accuracy >= 10)
225  {
226  beta = 0.05f;
227  }
